# liquidity_pipeline_fixed.py
import warnings
warnings.filterwarnings("ignore")

import numpy as np
import pandas as pd
from datetime import datetime, timedelta
from pandas_datareader import data as web

from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import ParameterGrid
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import classification_report, roc_auc_score, roc_curve
from sklearn.base import clone

from scipy.stats import chi2
from numpy.linalg import inv


try:
    from arch import arch_model
    HAS_ARCH = True
except:
    HAS_ARCH = False

# -------------------------
# Config
# -------------------------
plumbing_series = {
    "SOFR": "SOFR",
    "ONRRP": "RRPONTSYD",
    "RESBAL": "WRESBAL",
    "TGA": "WTREGEN"
}

market_series = {
    "EFFR": "EFFR",
    "Tbill": "DTB3",
    "VIX": "VIXCLS",
    "CorpSpread": "BAA_AAA"
}

start = datetime.today() - timedelta(days=900)
end = datetime.today()

# Mahalanobis threshold: percentile (e.g., 0.995 to flag extremes)
MAHA_PCT = 0.995

# Grid for ElasticNet-like tunable balance using LogisticRegression with 'saga' solver
GRID = {
    "C": [0.1, 1.0, 10.0],         # inverse regularization strength
    "l1_ratio": [0.0, 0.5, 1.0]    # 0 -> pure L2, 1 -> pure L1, 0.5 -> mix
}

# -------------------------
# Helpers
# -------------------------
def download_fred_series(series_dict, start, end):
    df = pd.concat(
        {name: web.DataReader(code, "fred", start, end).squeeze() for name, code in series_dict.items()},
        axis=1
    )
    df = df.ffill().bfill()
    return df

# Mahalanobis distance (multivariate)
def mahalanobis_dist(df):
    X = df.values
    mu = X.mean(axis=0)
    cov = np.cov(X, rowvar=False)
    # regularize cov if singular
    cov += np.eye(cov.shape[0]) * 1e-6
    invcov = inv(cov)
    diffs = X - mu
    m = np.einsum('ij,jk,ik->i', diffs, invcov, diffs)
    return m

# EWMA volatility fallback
def ewma_vol(series, span=30):
    return series.ewm(span=span).std()

# Fit GARCH with fallback to EWMA
def get_conditional_vol(series, try_garch=True):
    if try_garch and HAS_ARCH:
        try:
            am = arch_model(series.dropna(), vol='Garch', p=1, q=1, dist='normal')
            res = am.fit(disp='off', show_warning=False)
            vol = pd.Series(res.conditional_volatility, index=series.dropna().index)
            vol = vol.reindex(series.index).ffill().bfill()
            return vol
        except Exception as e:
            # fallback to EWMA
            return ewma_vol(series)
    else:
        return ewma_vol(series)

# Walk-forward expanding-window generator
def walk_forward_split(index, initial_train_size, test_size):
    n = len(index)
    start = 0
    train_end = initial_train_size
    while train_end + test_size <= n:
        train_idx = list(range(0, train_end))
        test_idx = list(range(train_end, train_end + test_size))
        yield train_idx, test_idx
        train_end += test_size  # sliding forward by test_size; you can change overlap

# -------------------------
# 1. Download data
# -------------------------
plumbing_data = download_fred_series(plumbing_series, start, end)
market_data = download_fred_series(market_series, start, end)
data = pd.concat([plumbing_data, market_data], axis=1)

# -------------------------
# 2. Feature engineering
# -------------------------
# daily changes for plumbing
for col in plumbing_series.keys():
    data[f'd{col}'] = data[col].diff()

# spreads
data['EFFR_SOFR'] = data['EFFR'] - data['SOFR']
data['Tbill_SOFR'] = data['Tbill'] - data['SOFR']

# drop raw base rates if you want (keep derived)
data = data.drop(columns=list(plumbing_series.keys()) + ['EFFR', 'Tbill'])

# drop first NaN row
data = data.iloc[1:].copy()

# -------------------------
# 3. Add lag1 features
# -------------------------
for col in list(data.columns):
    data[f'{col}_lag1'] = data[col].shift(1)

# -------------------------
# 4. Add volatility features (GARCH/EWMA) safely
# -------------------------
vol_targets = ['dSOFR', 'dONRRP', 'dRESBAL', 'dTGA', 'EFFR_SOFR', 'Tbill_SOFR']
for col in vol_targets:
    if col in data.columns:
        data[f'{col}_vol'] = get_conditional_vol(data[col], try_garch=True)

# -------------------------
# 5. Clean: drop rows with NaN after lags/vols
# -------------------------
data = data.dropna().copy()

# -------------------------
# 6. Outlier detection: Mahalanobis + winsorize
# -------------------------
# Choose numeric columns for multivariate test (exclude identifiers)
numeric_cols = data.select_dtypes(include=[np.number]).columns.tolist()
m = mahalanobis_dist(data[numeric_cols])
pvals = 1 - chi2.cdf(m, df=len(numeric_cols))
# flag extremes
extreme_mask = m > chi2.ppf(MAHA_PCT, df=len(numeric_cols))

# Option: drop extreme rows OR winsorize them
# We'll *keep* rows but winsorize variables at 1st/99th percentiles for robustness
lower_q = data[numeric_cols].quantile(0.01)
upper_q = data[numeric_cols].quantile(0.99)
data[numeric_cols] = data[numeric_cols].clip(lower=lower_q, upper=upper_q, axis=1)

# -------------------------
# 7. Standardize inputs (for modeling) — but keep a copy of raw-standardized for LSI
# -------------------------
scaler = StandardScaler()
data_scaled = pd.DataFrame(scaler.fit_transform(data[numeric_cols]), columns=numeric_cols, index=data.index)

# -------------------------
# 8. Construct LSI target using standardized building blocks
#    LSI = +1 * z(dSOFR)  -0.3 * z(dONRRP) -0.2 * z(dRESBAL) +0.3 * z(dTGA)
#    (We standardize first and then weight)
# -------------------------
components = {}
for term in ['dSOFR', 'dONRRP', 'dRESBAL', 'dTGA']:
    if term in data_scaled.columns:
        components[term] = data_scaled[term]

# if any missing, raise
if len(components) < 4:
    raise ValueError("Required LSI terms missing after preprocessing.")

lsi_series = (
    1.0 * components['dSOFR']
    - 0.3 * components['dONRRP']
    - 0.2 * components['dRESBAL']
    + 0.3 * components['dTGA']
)

# put LSI into the true (unscaled) frame for records (but it's standardized already)
data['LSI'] = lsi_series.values

# Define binary target by thresholding LSI (top 25% as 'tight' default)
threshold = data['LSI'].quantile(0.75)
data['y'] = (data['LSI'] > threshold).astype(int)

# -------------------------
# 9. Feature set selection initial: use intuitive plumbing+market slices
#    We'll take a reasonable set and let ElasticNet choose sparse subset
# -------------------------
candidate_features = [
    # primary movers
    'dONRRP', 'dTGA', 'dRESBAL', 'dSOFR',
    # lags
    'dONRRP_lag1', 'dTGA_lag1', 'dRESBAL_lag1', 'dSOFR_lag1',
    # vols
    'dONRRP_vol', 'dTGA_vol', 'dRESBAL_vol', 'dSOFR_vol',
    # spreads and market
    'EFFR_SOFR', 'Tbill_SOFR', 'VIX', 'CorpSpread',
    # lagged spreads
    'EFFR_SOFR_lag1', 'Tbill_SOFR_lag1'
]

# Keep only available ones
candidate_features = [f for f in candidate_features if f in data.columns]

X_all = data[candidate_features].copy()
y_all = data['y'].copy()

# standardize X_all (again) for model input, but use prior scaler fit on numeric columns optionally
X = pd.DataFrame(StandardScaler().fit_transform(X_all), columns=X_all.columns, index=X_all.index)

# -------------------------
# 10. Walk-forward CV + grid search (expanding window)
# -------------------------
initial_train_size = int(len(X) * 0.6)  # start with 60% as first training window
test_size = max(1, int(len(X) * 0.1))   # 10% for each test fold
results = []

best_global = None
best_score = -np.inf

# small param grid to limit runtime; we manually implement time-series CV
param_grid = list(ParameterGrid(GRID))

for params in param_grid:
    # logistic with elastic-net mix via l1_ratio and penalty='elasticnet'
    C = params['C']
    l1_ratio = params['l1_ratio']
    clf = LogisticRegression(
        penalty='elasticnet' if l1_ratio not in (0.0, 1.0) else ('l1' if l1_ratio==1.0 else 'l2'),
        solver='saga',
        C=C,
        l1_ratio=None if (l1_ratio in (0.0, 1.0)) else l1_ratio,
        class_weight='balanced',
        max_iter=2000,
        random_state=42
    )
    fold_scores = []
    # walk-forward
    for train_idx, test_idx in walk_forward_split(X.index, initial_train_size, test_size):
        X_train = X.iloc[train_idx]
        y_train = y_all.iloc[train_idx]
        X_test = X.iloc[test_idx]
        y_test = y_all.iloc[test_idx]

        # fit
        try:
            clf.fit(X_train, y_train)
            probs = clf.predict_proba(X_test)[:, 1]
            auc = roc_auc_score(y_test, probs)
        except Exception:
            auc = 0.5  # failed fit, penalize
        fold_scores.append(auc)
    mean_auc = np.mean(fold_scores)
    results.append((params, mean_auc))
    if mean_auc > best_score:
        best_score = mean_auc
        best_global = (params, mean_auc)

print("Grid search results (params -> mean ROC AUC):")
for p, s in results:
    print(p, f"{s:.4f}")
print("Best params:", best_global)

# -------------------------
# 11. Fit final model on full training up to last test start
# -------------------------
best_params = best_global[0]
C = best_params['C']
l1_ratio = best_params['l1_ratio']

final_clf = LogisticRegression(
    penalty='elasticnet' if l1_ratio not in (0.0, 1.0) else ('l1' if l1_ratio==1.0 else 'l2'),
    solver='saga',
    C=C,
    l1_ratio=None if (l1_ratio in (0.0, 1.0)) else l1_ratio,
    class_weight='balanced',
    max_iter=5000,
    random_state=42
)

# Train on initial_train_size + all subsequent test windows except the final holdout (simulate last-window evaluation)
train_idx, test_idx = list(walk_forward_split(X.index, initial_train_size, test_size))[-1]
X_train_full = X.iloc[:(train_idx[-1]+1)]
y_train_full = y_all.iloc[:(train_idx[-1]+1)]
X_test_final = X.iloc[test_idx]
y_test_final = y_all.iloc[test_idx]

final_clf.fit(X_train_full, y_train_full)
preds = final_clf.predict(X_test_final)
probs = final_clf.predict_proba(X_test_final)[:,1]

print("\nFinal fold classification report:")
print(classification_report(y_test_final, preds, digits=3))
print("Final fold ROC AUC:", roc_auc_score(y_test_final, probs))

# -------------------------
# 12. Feature importance (coeff) - selected features
# -------------------------
coefs = pd.Series(final_clf.coef_.ravel(), index=X.columns).sort_values(key=np.abs, ascending=False)
print("\nTop model coefficients (abs-sorted):")
print(coefs.head(20))

# -------------------------
# 13. Predict latest date
# -------------------------
latest_row = X.iloc[[-1]]
latest_prob = final_clf.predict_proba(latest_row)[:,1][0]
latest_pred = "TIGHT" if latest_prob > 0.5 else "LOOSE"
print(f"\nLatest date: {X.index[-1].strftime('%Y-%m-%d')} -> {latest_pred} (p={latest_prob:.3f})")
print(f"LSI latest: {data['LSI'].iloc[-1]:.3f} (threshold {threshold:.3f})")

# -------------------------
# 14. Save model artifacts (optional)
# -------------------------

# Model Diagnostic Tests: Calibration & Stability
# ============================================

import matplotlib.pyplot as plt
from sklearn.calibration import calibration_curve
from sklearn.metrics import brier_score_loss
from sklearn.utils import resample

# ---------------------------------------------------
# 1. Probability Calibration / Reliability Plot + Brier Score
# ---------------------------------------------------
probs_test = probs  # from final fold in your model
true_test = y_test_final

brier = brier_score_loss(true_test, probs_test)
print(f"\nBrier Score (lower = better calibration): {brier:.4f}")

prob_true, prob_pred = calibration_curve(true_test, probs_test, n_bins=10)

plt.figure(figsize=(6,6))
plt.plot(prob_pred, prob_true, "s-", label="Model calibration")
plt.plot([0,1], [0,1], "k--", label="Perfect calibration")
plt.xlabel("Predicted probability (tight liquidity)")
plt.ylabel("Observed frequency")
plt.title("Reliability Curve / Probability Calibration")
plt.legend()
plt.grid(True)
plt.show()

# ---------------------------------------------------
# 2. Bootstrap Confidence Intervals (AUC + Coefficients)
# ---------------------------------------------------
n_boot = 200
auc_scores = []
coef_samples = []

for _ in range(n_boot):
    Xb, yb = resample(X_train_full, y_train_full, random_state=_)
    model_b = clone(final_clf)
    model_b.fit(Xb, yb)
    probs_b = model_b.predict_proba(X_test_final)[:,1]
    auc_scores.append(roc_auc_score(y_test_final, probs_b))
    coef_samples.append(model_b.coef_.ravel())

auc_mean = np.mean(auc_scores)
auc_ci = np.percentile(auc_scores, [2.5, 97.5])
print(f"\nBootstrap AUC 95% CI: {auc_mean:.3f} [{auc_ci[0]:.3f}, {auc_ci[1]:.3f}]")

coef_samples = np.array(coef_samples)
coef_df = pd.DataFrame(coef_samples, columns=X.columns)
coef_summary = coef_df.describe(percentiles=[0.025, 0.5, 0.975]).T[['2.5%', '50%', '97.5%']]
print("\nBootstrap Coefficient Intervals (95%):")
print(coef_summary.head(10))

# ---------------------------------------------------
# 3. Rolling Coefficient Stability (Time Drift Test)
# ---------------------------------------------------
window = 180  # ~6 months of trading days
roll_dates = []
coef_time = []

for end_i in range(window, len(X_train_full)):
    Xr = X_train_full.iloc[:end_i]
    yr = y_train_full.iloc[:end_i]
    clf_r = clone(final_clf)
    clf_r.fit(Xr, yr)
    roll_dates.append(X_train_full.index[end_i])
    coef_time.append(clf_r.coef_.ravel())

coef_time = np.array(coef_time)
plt.figure(figsize=(10,6))
for i, f in enumerate(X.columns[:5]):  # show top 5 for readability
    plt.plot(roll_dates, coef_time[:,i], label=f)
plt.title("Rolling Coefficient Stability (first 5 features)")
plt.xlabel("Date")
plt.ylabel("Coefficient value")
plt.legend()
plt.grid(True)
plt.show()
